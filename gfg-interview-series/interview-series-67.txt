import math
class Solution:
    def minimumNumber(self, n, arr):
        def gcd(a, b):
            if b == 0:
                return abs(a)
            else:
                return gcd(b, a%b)
        g = arr[0]
        for i in range(1, n):
            g = math.gcd(g, arr[i])
        return g
